DECLARE @HOTEL_NAME VARCHAR(MAX)
DECLARE @ROOM_TYPE_NAME VARCHAR(MAX)
DECLARE @FROM_DATE DATETIME
DECLARE @TO_DATE DATETIME

SET @HOTEL_NAME = 'Amari Atrium'
SET @ROOM_TYPE_NAME = 'DLX'
SET @FROM_DATE = '11/01/2012'
SET @TO_DATE = '11/15/2012'
DECLARE @TEMP INT
SET @TEMP = '-1' 

SET @TEMP = (SELECT COUNT(*) FROM SUPPLIER_HOTEL_PRICE_LIST
WHERE SUPPLIER_SR_NO = (SELECT SUPPLIER_SR_NO FROM SUPPLIER_CONTACT_DETAILS WHERE CHAIN_NAME = @HOTEL_NAME) AND ROOM_TYPE_ID = (SELECT ROOM_TYPE_ID FROM FARE_HOTEL_ROOM_TYPE_MASTER WHERE ROOM_TYPE_NAME = @ROOM_TYPE_NAME) AND ((@FROM_DATE BETWEEN SUPPLIER_HOTEL_PRICE_LIST.FROM_DATE AND SUPPLIER_HOTEL_PRICE_LIST.TO_DATE)OR (@TO_DATE BETWEEN SUPPLIER_HOTEL_PRICE_LIST.FROM_DATE AND  SUPPLIER_HOTEL_PRICE_LIST.TO_DATE)))
		
PRINT @TEMP
